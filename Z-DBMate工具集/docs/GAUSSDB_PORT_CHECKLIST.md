## 端口检测命令

```bash
# 检查本地端口监听
netstat -tlnp | grep -E '8002|40080|2379|2380|9092|8086'

# 检查远程端口可达性
telnet 192.168.x.x 8002
nc -zv 192.168.x.x 40080

# 批量端口扫描
for port in 22 8002 40080 2379 2380 9092 8086; do
  timeout 2 bash -c "echo > /dev/tcp/192.168.x.x/$port" 2>/dev/null \
    && echo "[OK] $port" || echo "[FAIL] $port"
done

# 防火墙状态
systemctl status firewalld
iptables -L -n | grep -E '8002|40080'
```
